STORE "ombak" as variable

  kecepatan = kecepatan - 0
ELSE IF "ombak" EQUALS 1
  kecepatan = kecepatan - 7
ELSE IF "ombak" EQUALS 2
  kecepatan = kecepatan - 15

  kecepatan = 0
ELSE
  DISPLAY "Masukkan bilangan bulat"
END IF
